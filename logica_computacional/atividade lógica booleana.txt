    private void btnAddActionPerformed(java.awt.event.ActionEvent evt) {                                       
        String elemento = campoItem.getText();//Obter Texto
        if(verificaSeExiste(elemento) == true){
            JOptionPane.showMessageDialog(this,"Este item já existe!");
        }
        else{
            lista.add(elemento);// adicionar na lista
            campoItem.setText(null);//Limpar o campo de texto
            exibeLista();
        }        
    }                                      

    private void btnRmvActionPerformed(java.awt.event.ActionEvent evt) {                                       
        String elemento = campoItem.getText();//Obter Texto
        if(verificaSeExiste(elemento) == true){
            lista.remove(elemento);
        }
        else{
            JOptionPane.showMessageDialog(this, "Este item não exite!");
        }
        lista.remove(elemento);// Remover da lista
        campoItem.setText(null);//Limpar o campo de texto
        exibeLista();
    }                                      

    private void btnEditActionPerformed(java.awt.event.ActionEvent evt) {                                        
        String elemento = campoItem.getText();
        if(verificaSeExiste(elemento) == true){
            lista.remove(elemento);
            lista.add(JOptionPane.showInputDialog(this, "Insira o novo item: "));
        }
        else{
            JOptionPane.showMessageDialog(this, "Este item não exite!");
        }
        exibeLista();
    }                                       
        public void exibeLista(){
            String html = "<html>Lista de Itens<br><br>";
            for(String item : lista){
                html += item+"<br>";
            }
            resposta.setText(html);
        }
        public boolean verificaSeExiste(String palavra){
            for(String item : lista){//Para cada item contido na lista
                if(item.equals(palavra)){//se o item for igual à palavra
                    return true;
                }
            }
            return false;
        }
